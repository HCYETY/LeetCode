## 二分查找
```c++
// 搜索左侧边界
int left_search(vector<int>&nums, int target) {
  int left = 0, right = nums.size() - 1;
  while(left <= right) {
    int mid = left + (right - left) / 2;
    if (nums[mid] == target) {
      right = mid - 1;
    } else if (nums[mid] > target) {
      right = mid - 1;
    } else if (nums[mid] < target) {
      left = mid + 1;
    }
  }
  if (left >= nums.size() || nums[left] != target) {
    return -1;
  }
  return left;
}
// 搜索右侧边界
int right_search(vector<int>&nums, int target) {
  int left = 0, right = nums.size() - 1;
  while(left <= right) {
    int mid = left + (right - left) / 2;
    if (nums[mid] == target) {
      left = mid + 1;
    } else if (nums[mid] > target) {
      right = mid - 1;
    } else if (nums[mid] < target) {
      left = mid + 1;
    }
  }
  if (right < 0 || nums[right] != target) {
    return -1;
  }
  return right;
}
```

## 前缀和模板
- 使用场景：原始数组不会被修改的情况下，频繁查询某个区间的累加和。
```c++
vector<int> ans;

void PrefixSum(vector<int>& nums) {
  ans.resize(nums.size() + 1);
  
  for (int i = 1; i < ans.size(); i++) {
    ans[i] = ans[i - 1] + nums[i - 1];
  }
}

/* 查询闭区间 [i, j] 的累加和 */
int query(int i, int j) {
  return ans[j + 1] - ans[i];
}
```
## 差分数组
- 使用场景：频繁对原始数组的某个区间的元素进行增减。
```c++
class Solution {
public:
  vector<int> diff;

  bool carPooling(vector<vector<int>>& trips, int capacity) {
    vector<int> nums(1001);
    difference(nums);

    for (vector<int> trip : trips) {
      int val = trip[0];
      int i = trip[1];
      int j = trip[2] - 1;
      increment(i, j, val);
    }

    vector<int> ans = result();
    for (int x : ans) {
      if (capacity < x) {
        return false;
      }
    }
    
    return true;
  }

  void difference(vector<int> nums) {
    diff.resize(nums.size());
    diff[0] = nums[0];
    for (int i = 1; i < nums.size(); i++) {
      diff[i] = nums[i] - nums[i - 1];
    }
  }
  void increment(int i, int j, int val) {
    diff[i] += val;
    if (j + 1 < diff.size()) {
      diff[j + 1] -= val;
    }
  }
  vector<int> result() {
    vector<int> ans(diff.size());
    ans[0] = diff[0];
    for (int i = 1; i < diff.size(); i++) {
      ans[i] = ans[i - 1] + diff[i];
    }
    return ans;
  }
};
```
